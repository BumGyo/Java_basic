AWT 그래픽 라이브러리를 사용해서 학생 성적을 등록할 수 있는 화면 만들기
핸드폰에서 친구 연락처를 새로 등록하기

예) // 변수는 하나의 데이터(값)만 저장
자료형 + 변수이름 = 값;
// 사람의 나이를 보관할 수 있는 변수 선언 + 초기화
int age = 20;
// 배열을 사용해서 최대 10명의 나이를 보관하기
int ages[] = new int[10];
// for 반복문을 사용해서 최대 10명의 사람의 나이를 사용자로부터 입력 받은
// 다음에 위에서 선언한 ages 배열에 저장합니다!!
java.util.Scanner inputScanner = new java.util.Scanner(System.in);
// 사용자로부터 입력 받은 나이를 정수로 보관할 임시 변수를 선언
int tempAge = 0;
for(int i = 0; i < ages.length; i++) {
    System.out.print(i + " 위치에 보관할 사람의 나이를 입력하세요: ");
    tempAge = inputScanner.nextInt( );
    ages[i] = tempAge;
    System.out.println("사용자가 입력한 사람의 나이는 " + ages[i] + 
      " 입니다!!");
}
// -> 메모리 공간을 사용하는데에 데이터의 갯수를 무한대로 확장해 주는 도구가
// -> 컬렉션(Collection) 입니다!!
// -> 학생 성적을 처리(등록을 저장)해주는 예제를 구현하기
// -> 1) 콘솔(Console) 형식: 키보드와 모니터)만 사용
// -> 2) GUI 형식(키보드, 마우스, 모니터)으로 만들겠습니다!!
